```bash

python -m venv venv
venv\Scripts\activate

# 2. Dependencies
pip install -r requirements.txt

# 3. MySQL Database
mysql -u root -p shop_db < database/shop_db_mysql.sql

# 4. Setup .env
# → Edit .env faylini MySQL ma'lumotlari bilan

# 5. Run Bot
python main.py

# 6. Test in Telegram
# → Send /start to @YourBotName
```
